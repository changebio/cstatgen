173  // we reverse the two bits as defined in PLINK format,
174  // so we can process 2-bit at a time.
175  const static unsigned char HOM_REF = 0x0; //0b00;
176  const static unsigned char HET = 0x2; //0b10;
177  const static unsigned char HOM_ALT = 0x3; //0b11;
178  const static unsigned char MISSING = 0x1; //0b01;
179 
180  if (mode == 0x01) {
181  // snp major mode
182  unsigned char mask[] = { 0x3, 0xc, 0x30, 0xc0 }; //0b11, 0b1100, 0b110000, 0b11000000
183  unsigned char c;
184  int numPeople = this->people2Idx.size();
185  int numMarker = this->marker2Idx.size();
186  (*this->genotype).Dimension( numMarker, numPeople);
187  for (int m = 0; m < numMarker; m++){
188  for (int p = 0; p < numPeople; p++) {
189  int offset = p & (4 - 1);
190  if (offset == 0) {
191  fread(&c, sizeof(unsigned char), 1, fBed);
192  }
193  unsigned char geno = (c & mask[offset]) >> (offset << 1);
194  switch (geno){
195  case HOM_REF:
196  (*this->genotype)[m][p] = 0;
197  break;
198  case HET:
199  (*this->genotype)[m][p] = 1;
200  break;
201  case HOM_ALT:
202  (*this->genotype)[m][p] = 2;
203  break;
204  case MISSING:
205  (*this->genotype)[m][p] = MISSING_GENOTYPE;
206  break;
207  default:
208  REPORT("Read PLINK genotype error!\n");
209  break;
210  };
211  }
212  }
213  } else if (mode == 0x00) {
214  // people major mode
215  // TODO
216  fprintf(stderr, "Cannot open BED file %s, work to be DONE.\n", prefix);
217  } else {
218  fprintf(stderr, "Cannot open BED file %s, unrecognized major mode.\n", prefix);
219  };
